```python
def determine_interests(docs: list) -> list:
    """从文档标题和内容提取兴趣领域"""
    # 1. 提取关键词
    keywords = []
    for doc in docs:
        # 从标题提取
        keywords.extend(extract_keywords(doc["title"]))
        # 从 frontmatter.description 提取
        if doc.get("description"):
            keywords.extend(extract_keywords(doc["description"]))
    
    # 2. 聚类并排序
    interest_clusters = cluster_keywords(keywords)
    
    # 3. 返回 top-N
    return [c["name"] for c in interest_clusters[:5]]

def assess_skill_level(docs: list, skill: str) -> str:
    """评估某技能的水平"""
    # 文档类型判断
    tutorial_count = count_docs_containing(docs, ["教程", "入门", "基础"])
    project_count = count_docs_containing(docs, ["项目", "实践", "实现"])
    advanced_count = count_docs_containing(docs, ["原理", "源码", "架构"])
    
    # 规则判断
    if advanced_count > 2:
        return "advanced"
    elif project_count > 3:
        return "intermediate"
    elif tutorial_count > 0:
        return "beginner"
    else:
        return "beginner"

def assess_collaboration_style(docs: list) -> str:
    """评估协作风格"""
    # 检查 frontmatter 中的 collaborators 字段
    collab_docs = [d for d in docs if d.get("collaborators")]
    
    if len(collab_docs) / len(docs) > 0.3:
        return "collaborative"
    else:
        return "solo"
```

## 5. 伙伴推荐

### 5.1 匹配维度

| 维度 | 权重 | 说明 |
|------|:----:|------|
| 兴趣重合度 | 40% | 共同感兴趣的领域 |
| 水平相近度 | 30% | 学习伙伴优先同级 |
| 活跃时间 | 15% | 最近活跃的人 |
| 协作风格 | 15% | solo 找 solo，collab 找 collab |

### 5.2 匹配算法

```python
def calculate_match_score(user_a: dict, user_b: dict) -> float:
    """计算两个用户的匹配分数"""
    
    # 1. 兴趣重合度 (40%)
    interests_a = set(user_a["profile"]["interests"])
    interests_b = set(user_b["profile"]["interests"])
    if interests_a and interests_b:
        interest_score = len(interests_a & interests_b) / len(interests_a | interests_b)
    else:
        interest_score = 0
    
    # 2. 水平相近度 (30%)
    level_map = {"beginner": 1, "intermediate": 2, "advanced": 3}
    level_diff = abs(level_map.get(user_a["profile"]["level"], 1) - 
                     level_map.get(user_b["profile"]["level"], 1))
    level_score = 1 - (level_diff / 2)  # 0-1 分，差距越小越高
    
    # 3. 活跃时间 (15%)
    # 最近 7 天有活动 = 1，否则按距离衰减
    active_score = calculate_active_score(user_b["last_active"])
    
    # 4. 协作风格 (15%)
    style_match = (user_a["profile"]["collaboration_style"] == 
                   user_b["profile"]["collaboration_style"])
    style_score = 1 if style_match else 0.5
    
    # 加权总分
    total = (interest_score * 0.4 + 
             level_score * 0.3 + 
             active_score * 0.15 + 
             style_score * 0.15)
    
    return round(total, 2)

def recommend_partners(user_id: int, scene: str = "learning", limit: int = 5) -> list:
    """推荐伙伴"""
    user_profile = load_profile(user_id)
    all_profiles = load_all_profiles()
    
    candidates = []
    for other in all_profiles:
        if other["yuque_id"] == user_id:
            continue
        
        score = calculate_match_score(user_profile, other)
        
        # 根据场景调整
        if scene == "mentor":
            # 导师：水平高一级优先
            if other["profile"]["level"] > user_profile["profile"]["level"]:
                score += 0.2
        
        candidates.append({
            "user": other,
            "score": score,
            "match_reasons": explain_match(user_profile, other)
        })
    
    # 按分数排序
    candidates.sort(key=lambda x: x["score"], reverse=True)
    return candidates[:limit]

def explain_match(user_a: dict, user_b: dict) -> list:
    """解释匹配原因"""
    reasons = []
    
    # 兴趣重合
    common = set(user_a["profile"]["interests"]) & set(user_b["profile"]["interests"])
    if common:
        reasons.append(f"共同兴趣：{', '.join(list(common)[:3])}")
    
    # 水平相近
    if user_a["profile"]["level"] == user_b["profile"]["level"]:
        reasons.append("水平相近，适合一起学习")
    elif user_b["profile"]["level"] > user_a["profile"]["level"]:
        reasons.append("经验更丰富，可以做 mentor")
    
    return reasons
```

### 7.2 阅读时长估算

**估算公式**：

```python
def estimate_reading_time(doc: dict) -> int:
    """估算阅读时长（分钟）"""
    # 1. 基于字数
    char_count = len(doc.get("content", ""))
    base_minutes = char_count / 500  # 假设 500 字/分钟
    
    # 2. 基于内容类型调整
    if "代码" in doc.get("content", ""):
        base_minutes *= 1.5  # 含代码阅读更慢
    
    if "公式" in doc.get("content", ""):
        base_minutes *= 1.3  # 含公式阅读更慢
    
    # 3. 最小最大限制
    return max(1, min(30, int(base_minutes)))

def estimate_ask_time(question: str) -> int:
    """估算问答时长"""
    # 简单问题 2 分钟，复杂问题 5 分钟
    if len(question) > 50:
        return 5
    return 2
```

```python
def aggregate_daily_learning(user_id: int, date: str) -> dict:
    """聚合某天的学习数据"""
    history = load_learning_history(user_id)
    daily = [h for h in history if h["date"] == date]
    
    total_minutes = 0
    topics = {}
    
    for h in daily:
        if h["action"] == "read":
            minutes = estimate_reading_time(get_doc(h["doc_id"]))
        elif h["action"] == "ask":
            minutes = estimate_ask_time(h.get("question", ""))
        else:
            minutes = 1
        
        total_minutes += minutes
        topic = h.get("topic", "其他")
        topics[topic] = topics.get(topic, 0) + minutes
    
    return {
        "date": date,
        "total_minutes": total_minutes,
        "topics": topics,
        "actions": {
            "read": len([h for h in daily if h["action"] == "read"]),
            "ask": len([h for h in daily if h["action"] == "ask"])
        }
    }
```
