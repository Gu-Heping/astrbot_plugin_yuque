## 5. 语雀 API 封装

```python
# api/yuque.py
import httpx
from typing import Dict, List, Optional

class YuqueAPI:
    """语雀 API 封装，独立实现，不依赖外部服务"""
    
    def __init__(self, token: str, base_url: str = "https://nova.yuque.com/api/v2"):
        self.token = token
        self.base_url = base_url.rstrip("/")
        self.headers = {
            "X-Auth-Token": token,
            "User-Agent": "NovaBot/1.0",
        }
        self.client = httpx.AsyncClient(headers=self.headers, timeout=30.0)
    
    async def get_user_info(self) -> Dict:
        """获取当前认证用户信息"""
        resp = await self.client.get(f"{self.base_url}/user")
        resp.raise_for_status()
        return resp.json().get("data", {})
    
    async def get_user_repos(self, user_id: int) -> List[Dict]:
        """获取用户的知识库列表"""
        resp = await self.client.get(f"{self.base_url}/users/{user_id}/repos")
        resp.raise_for_status()
        return resp.json().get("data", [])
    
    async def get_repo_docs(self, repo_id: int) -> List[Dict]:
        """获取知识库文档列表"""
        resp = await self.client.get(f"{self.base_url}/repos/{repo_id}/docs")
        resp.raise_for_status()
        return resp.json().get("data", [])
    
    async def get_doc_detail(self, repo_id: int, slug: str) -> Optional[Dict]:
        """获取文档详情"""
        resp = await self.client.get(f"{self.base_url}/repos/{repo_id}/docs/{slug}")
        if resp.status_code == 404:
            return None
        resp.raise_for_status()
        return resp.json().get("data", {})
    
    async def search_docs(self, query: str, repo_ids: List[int] = None) -> List[Dict]:
        """搜索文档（语雀搜索 API）"""
        params = {"q": query}
        if repo_ids:
            params["repo_ids"] = ",".join(str(rid) for rid in repo_ids)
        
        resp = await self.client.get(f"{self.base_url}/search", params=params)
        resp.raise_for_status()
        return resp.json().get("data", {}).get("items", [])
```

## 6. 核心工具实现

### 6.1 search_docs - 检索文档

```python
@dataclass
class SearchDocsTool(FunctionTool):
    name: str = "search_docs"
    description: str = "检索语雀知识库文档"
    parameters: dict = field(default_factory=lambda: {
        "type": "object",
        "properties": {
            "query": {
                "type": "string",
                "description": "搜索关键词"
            },
            "repo": {
                "type": "string",
                "description": "知识库名称（可选）"
            }
        },
        "required": ["query"]
    })
    
    async def run(self, event, query: str, repo: str = None):
        """
        1. 调用语雀搜索 API
        2. 返回文档列表 + 摘要
        """
        from api.yuque import YuqueAPI
        from utils.storage import get_config
        
        config = get_config()
        client = YuqueAPI(config["yuque_token"], config.get("yuque_base_url"))
        
        # 搜索
        results = await client.search_docs(query)
        
        # 格式化输出
        if not results:
            return {"found": False, "message": "未找到相关文档"}
        
        docs = []
        for item in results[:10]:
            docs.append({
                "title": item.get("title", ""),
                "author": item.get("author", {}).get("name", ""),
                "url": f"https://nova.yuque.com/{item.get('path', '')}",
                "summary": item.get("description", "")[:200] if item.get("description") else "",
            })
        
        return {"found": True, "docs": docs, "count": len(docs)}
```

@dataclass
class EvaluateEngagementTool(FunctionTool):
    name: str = "evaluate_engagement"
    description: str = "评估成员的真实参与质量"
    parameters: dict = field(default_factory=lambda: {
        "type": "object",
        "properties": {
            "user_id": {
                "type": "string",
                "description": "用户标识"
            },
            "period": {
                "type": "string",
                "description": "评估周期",
                "default": "2weeks"
            }
        },
        "required": ["user_id"]
    })
    
    async def run(self, event, user_id: str, period: str = "2weeks"):
        """
        综合评估参与质量：
        1. 文档质量（真实思考 vs 合规痕迹）
        2. 项目推进（里程碑）
        3. 协作贡献
        4. 参与形式多样性
        """
        from services.engagement import EngagementService
        
        service = EngagementService()
        score = await service.evaluate(user_id, period)
        
        return {
            "user_id": user_id,
            "period": period,
            "engagement_score": score.total,
            "breakdown": {
                "doc_quality": score.doc_quality,
                "project_progress": score.project_progress,
                "collaboration": score.collaboration,
                "diversity": score.diversity,
            },
            "warning": score.warning if score.total < 40 else None,
        }

## 10. 与现有系统的关系

| 系统 | 定位 | NovaBot 如何使用 |
|:-----|:-----|:------------------|
| yuque2git | 语雀同步服务 | 不直接使用。NovaBot 通过语雀 API 自己获取数据 |
| gno MCP | OpenClaw 的检索能力 | 不直接使用。NovaBot 有自己的 search_docs 工具 |
| OpenClaw | Agent 平台 | NovaBot 是 AstrBot 插件，运行在 AstrBot 上，可以与 OpenClaw 通信 |

**设计原则**：
- 独立、轻量、无硬耦合
- 用户配置 Token → NovaBot 自己调用 API
- 核心价值在业务逻辑，不在基础设施
