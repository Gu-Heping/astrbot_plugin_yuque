## 1. 设计原则

### 1.1 来自官方的设计原则

| 原则 | 说明 |
|------|------|
| **陪伴感** | 不是工具，是伙伴 |

## 5. 技术实现要点

### 5.1 情绪识别

**识别方式**：

| 方式 | 准确度 | 成本 | 推荐场景 |
|------|:------:|:----:|----------|
| 关键词匹配 | 低 | 零 | MVP |
| LLM 分析 | 高 | 有 | 生产环境 |
| 混合方案 | 中 | 低 | 推荐 |

**关键词匹配（MVP）**：

```python
EMOTION_KEYWORDS = {
    "negative": {
        "累": 0.8, "烦": 0.7, "难": 0.6, "不想": 0.7, "放弃": 0.9,
        "压力": 0.8, "焦虑": 0.9, "崩溃": 1.0, "抑郁": 1.0,
        "迷茫": 0.7, "困惑": 0.5, "卡住了": 0.6
    },
    "positive": {
        "开心": 0.8, "兴奋": 0.9, "终于": 0.7, "搞定": 0.8,
        "成功": 0.7, "进步": 0.6, "理解了": 0.7, "学会了": 0.8
    },
    "neutral": ["嗯", "好的", "明白", "知道了"]
}

def detect_emotion_simple(text: str) -> dict:
    """简单关键词匹配"""
    score = {"positive": 0, "negative": 0, "neutral": 0}
    
    for emotion, keywords in EMOTION_KEYWORDS.items():
        if emotion == "neutral":
            continue
        for kw, weight in keywords.items():
            if kw in text:
                score[emotion] += weight
    
    # 归一化
    total = score["positive"] + score["negative"]
    if total > 0:
        score["positive"] /= total
        score["negative"] /= total
    
    # 判断主导情绪
    if score["negative"] > 0.6:
        return {"emotion": "negative", "confidence": score["negative"]}
    elif score["positive"] > 0.6:
        return {"emotion": "positive", "confidence": score["positive"]}
    else:
        return {"emotion": "neutral", "confidence": 0.5}
```

**LLM 分析（推荐）**：

```python
EMOTION_PROMPT = """
分析用户消息中的情绪状态。

消息：{message}

请判断：
1. 情绪类型：positive / negative / neutral / stressed / frustrated
2. 强度：0.0 - 1.0
3. 建议：是否需要特殊处理

以 JSON 返回：
{
  "emotion": "类型",
  "intensity": 0.7,
  "needs_support": true,
  "suggested_response_tone": "温柔、鼓励"
}
"""

async def detect_emotion_llm(message: str, llm_provider) -> dict:
    """用 LLM 分析情绪"""
    prompt = EMOTION_PROMPT.format(message=message)
    result = await llm_provider.text_chat(prompt, context=[])
    return parse_json(result.completion_text)
```

RESPONSE_TEMPLATES = {
    "negative": {
        "high_intensity": {
            "prefix": "……我听到了。",

            "tone": "共情、轻声",
            "suggest": "如果有具体问题，我可以帮你看看。"
        }
    },
    "stressed": {
        "prefix": "……最近压力挺大？",
        "tone": "理解、不评判",
        "suggest": "文档不会跑，慢慢来。"
    },
    "frustrated": {
        "prefix": "卡住了？",
        "tone": "平静、帮忙",
        "suggest": "说出来我帮你想想办法。"
    }
}

def generate_emotion_aware_response(emotion: dict, base_response: str) -> str:
    """根据情绪调整回复"""
    if emotion["emotion"] == "negative" and emotion.get("intensity", 0) > 0.7:
        template = RESPONSE_TEMPLATES["negative"]["high_intensity"]
        return f"{template['prefix']}\n{template['suggest']}"
    
    return base_response
```

    history = load_emotion_history(user_id)
    history.append({
        "timestamp": datetime.now().isoformat(),

    })

def detect_emotion_trend(user_id: int, days: int = 7) -> dict:
    """检测情绪趋势"""
    history = load_emotion_history(user_id)
    recent = [h for h in history if is_within_days(h["timestamp"], days)]
    
    if not recent:
        return {"trend": "unknown"}
    
    negative_ratio = len([h for h in recent if h["emotion"] == "negative"]) / len(recent)
    
    if negative_ratio > 0.6:
        return {"trend": "concerning", "suggestion": "可能需要关注"}
    elif negative_ratio > 0.3:
        return {"trend": "fluctuating", "suggestion": "情绪波动"}
    else:
        return {"trend": "stable", "suggestion": "状态良好"}
```
